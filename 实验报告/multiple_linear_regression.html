def loadDataSet(fileName):

    """
    :

def lwlr(testPoint, xArr, yArr, k=1.0):
    """
    
    parameters:
        testPoint -
        xArr -
        yArr -
        k -k
    return:
        testPoint * ws -
    """
    #
    xMat = mat(xArr); yMat = mat(yArr).T
    #
    m = shape(xMat)[0]
    weights = mat(eye((m)))  #m*m
    for j in range(m):                             #
        #
        diffMat =testPoint-xMat[j, :]  
        #
        weights[j, j] = exp(diffMat * diffMat.T/(-2.0 * k**2))
    #
    xTx = xMat.T * (weights * xMat)                   #m*m                     
    #0
    if linalg.det(xTx) == 0.0:
        return
    ws = xTx.I * (xMat.T * (weights * yMat))          #m*1
    #
    return testPoint * ws


def lwlrTest(testArr, xArr, yArr,k=1.0):
    """
    
    parameters:
        testArr -
        xArr -
        yArr -
        k -k
    return:
        yHat -
    """
    #
    m=np.shape(testArr)[0]
    yHat=np.zeros(m)#
    for i in range(m):
        yHat[i]=lwlr(testArr[i],xArr,yArr,k)
    return yHat
