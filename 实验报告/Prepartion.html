def file2matrix(filename):
    """
    
    parameters:
        fileName - 
    return:
        featureMat - 
        classLabelVector - (didntLike - 0, smallDoses - 1, largeDoses - 2)
    """
    # 
    fr = open(filename)
    # 
    arrayOLines = fr.readlines()
    # 
    numberOfLines = len(arrayOLines)
    # NumPy,:numberOfLines,3
    returnMat = np.zeros((numberOfLines,3))
    print(returnMat)
    # 
    classLabelVector = []
    # 
    index = 0
    for line in arrayOLines:
        # s.strip(rm)rm,('\n','\r','\t',' ')
        line = line.strip()
        # s.split(str="",num=string,cout(str))'\t'
        listFormLine = line.split('\t')
        # ,returnMatNumPy,
        returnMat[index,:] = listFormLine[0:3]
        # ,1,2,3
        if listFormLine[-1] == 'didntLike':
            classLabelVector.append(1)
        elif listFormLine[-1] == 'smallDoses':
            classLabelVector.append(2)
        elif listFormLine[-1] == 'largeDoses':
            classLabelVector.append(3)
        index += 1

    return returnMat,classLabelVector

def showdatas(datingDataMat, datingLabels):
    """
    
    parameters:
        datingDataMat - 
        datingLabels - (didntLike - 0, smallDoses - 1, largeDoses - 2)
    """
    # 
    labels = ['didntLike', 'smallDoses', 'largeDoses']
    features = ['Number of frequent flyer miles earned per year', 
              'Percentage of time spent playing video games',
              'Liters of ice cream consumed per week']
    fig, ax = plt.subplots(3, figsize = (12, 12))
    # 
    colornames = ['black', 'orange', 'red']
    # dataLabels
    colorv = [colornames[x-1] for x in datingLabels]
    # 
    scatter0 = ax[0].scatter(x=datingDataMat[:,0], y=datingDataMat[:,1], c=colorv, s=15, alpha=0.5, label=labels)
    ax[0].set_xlabel(features[0])
    ax[0].set_ylabel(features[1])
    ax[1].scatter(x=datingDataMat[:,0], y=datingDataMat[:,2], c=colorv, s=15, alpha=0.5, label=labels)
    ax[1].set_xlabel(features[0])
    ax[1].set_ylabel(features[2])
    ax[2].scatter(x=datingDataMat[:,1], y=datingDataMat[:,2], c=colorv, s=15, alpha=0.5, label=labels)
    ax[2].set_xlabel(features[1])
    ax[2].set_ylabel(features[2])
    # 
    didntLike = mlines.Line2D([], [], color='black', marker='.',
                              markersize=6, label='didntLike')
    smallDoses = mlines.Line2D([], [], color='orange', marker='.',
                               markersize=6, label='smallDoses')
    largeDoses = mlines.Line2D([], [], color='red', marker='.',
                               markersize=6, label='largeDoses')
    # 
    ax[0].legend(handles=[didntLike, smallDoses, largeDoses])
    ax[1].legend(handles=[didntLike, smallDoses, largeDoses], loc='lower right')
    ax[2].legend(handles=[didntLike, smallDoses, largeDoses], loc='upper right')
    plt.show()
